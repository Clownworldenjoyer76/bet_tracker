    "home": float(row.get("home_edge_decimal",0)),
    "away": float(row.get("away_edge_decimal",0))
}

edge = edges[side]

venue = side
odds = float(row.get(f"{side}_dk_american",0))
fav_ud = "favorite" if odds < 0 else "underdog"


# -----------------------------
# STRUCTURAL MARKET FILTERS
# -----------------------------

# Fade overpriced home favorites
if fav_ud == "favorite" and venue == "home" and -180 <= odds <= -150:
    continue

# Avoid massive favorites
if fav_ud == "favorite" and odds <= -500:
    continue

# Avoid extreme longshot underdogs
if fav_ud == "underdog" and odds >= 350:
    continue


# -----------------------------
# EDGE REQUIREMENTS
# -----------------------------

if fav_ud == "favorite":
    edge_required = 0.08
else:
    edge_required = 0.07

# Reduce exposure to home favorites
if fav_ud == "favorite" and venue == "home":
    edge_required = 0.10

# Boost away underdogs
if fav_ud == "underdog" and venue == "away" and 130 <= odds <= 170:
    edge_required = 0.06

# Home court penalty
if venue == "home":
    edge_required += 0.02

if edge < edge_required:
    continue
